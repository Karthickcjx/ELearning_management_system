+    private String normalizeMediaSignalType(String type) {
+        if (type == null) {
+            return null;
+        }
+
+        String normalized = type.trim().toUpperCase(Locale.ROOT);
+        if (normalized.isEmpty()) {
+            return null;
+        }
+
+        if (normalized.length() > MAX_SIGNAL_TYPE_LENGTH) {
+            normalized = normalized.substring(0, MAX_SIGNAL_TYPE_LENGTH);
+        }

+        if (!ALLOWED_MEDIA_SIGNAL_TYPES.contains(normalized)) {
+            throw new IllegalArgumentException("Unsupported room media signal type");
+        }

+        return normalized;
+    }
